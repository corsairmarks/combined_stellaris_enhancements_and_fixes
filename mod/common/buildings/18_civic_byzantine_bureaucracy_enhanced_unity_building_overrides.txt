	base_buildtime = @b3_time
	can_build = no

	category = unity

	potential = {

			is_spiritualist = no
			has_make_spiritualist_perk = no
		}
		NOT = { has_modifier = slave_colony }
	}

		has_upgraded_capital = yes
	}

	destroy_trigger = {
		owner = {
			OR = {
				is_regular_empire = no
				is_spiritualist = yes
				has_make_spiritualist_perk = yes
			}
		}
	}

			minerals = @b2_minerals
			rare_crystals = @b3_rare_cost
		}
		upkeep = {
			energy = @b3_upkeep
			rare_crystals = @b3_rare_upkeep
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_megacorp = no }
		}
		modifier = {
			job_bureaucrat_add = @b3_jobs
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_megacorp = no }
		}
		text = job_bureaucrat_effect_desc
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner.federation
			owner.federation = { has_federation_perk = all_the_priests_2 }
		}

	}

	triggered_desc = {
		trigger = {
			exists = owner.federation

		}
		text = job_high_priest_effect_desc
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_valid_civic = civic_byzantine_bureaucracy }
		}
		modifier = {
			job_bureaucrat_add = -1
			job_magistrate_add = 1
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { has_valid_civic = civic_byzantine_bureaucracy }
		}
		text = job_magistrate_effect_desc
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_megacorp = yes }
		}

			job_manager_add = @b3_jobs
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_megacorp = yes }
		}
		text = job_manager_effect_desc
	}

		"tech_galactic_bureaucracy"
	}

	show_tech_unlock_if = {
		is_regular_empire = yes
		is_spiritualist = no
		has_make_spiritualist_perk = no
