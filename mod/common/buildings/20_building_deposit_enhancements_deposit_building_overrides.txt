	}
	triggered_planet_modifier = {

		planet_technician_energy_produces_add = 2
	}
	triggered_planet_modifier = {
		potential = {

	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		job_technician_drone_add = @b2_jobs

	triggered_planet_modifier = {
		potential = {
			NOR = {
				is_planet_class = pc_habitat

				is_planet_class = pc_hive

			}
		}
		district_generator_max = 4
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_technician_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_technician_effect_desc
	}

		"tech_mine_betharian"
	}
}

# Modified:
# * +1 culture worker

# * +25% culture worker output
# * culture worker job description
building_xeno_zoo = {
	base_buildtime = @b1_time
	is_capped_by_modifier = yes
	category = resource
	potential = {
		exists = owner
		owner = {
			is_regular_empire = yes
		}
	}
	destroy_trigger = {
		exists = owner
		owner = { is_regular_empire = no }
	}
	resources = {
		category = planet_buildings
		cost = {
			minerals = 200
		}
		upkeep = {
			energy = 1
		}
	}
	planet_modifier = {
		job_entertainer_add = 2
		job_culture_worker_add = 1
		job_culture_worker_per_pop = 0.04 # 1 per 25
		planet_culture_workers_produces_mult = 0.25 # high boost for a rare job

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_valid_civic = civic_warrior_culture }
		}
		job_duelist_add = 2
		job_entertainer_add = -2
	}
	prerequisites = {
		"tech_alien_life_studies"
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { has_valid_civic = civic_warrior_culture }
		}
		text = job_duelist_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { NOT = { has_valid_civic = civic_warrior_culture } }
		}
		text = job_entertainer_effect_desc
	}
	triggered_desc = {
		text = job_culture_worker_effect_desc
	}
}
