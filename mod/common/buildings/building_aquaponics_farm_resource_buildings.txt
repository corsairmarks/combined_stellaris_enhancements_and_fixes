building_aquaponics_farm = {
	base_buildtime = @b1_time # intentionally retains b1 build-time
	can_build = no
	category = resource
	potential = {
		exists = owner
		owner = {
			OR = {
				country_uses_food = yes
				is_ai = no
			}
		}
		NOR = {
			is_planet_class = pc_machine
			has_modifier = resort_colony
		}
	}

	allow = {
	}

	destroy_trigger = {
		exists = owner
		owner = {
			is_ai = yes
			OR = {
				country_uses_food = no
				AND = {
					has_monthly_income = {
						resource = food
						value > 100
					}
					prev = {
						NOT = { is_planet_class = pc_habitat }
						free_building_slots = 0
					}
				}
			}
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			job_farmer_add = 1
			job_angler_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_agri_drone_add = 1
			job_aquaculture_drone_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_fallen_empire_spiritualist = yes }
		}
		modifier = {
			job_farmer_add = -1
			job_fe_acolyte_farm_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_origin = origin_void_dwellers }
		}
		modifier = {
			job_farmer_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
				is_anglers_empire = yes
				prev = {
					OR = {
						is_wet = yes
						has_modifier = flooded_habitat
					}
				}
			}
		}
		modifier = {
			job_angler_add = 1
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals # intentionally retains b1 cost
		}
		upkeep = {
			energy = @b1_upkeep # intentionally retains b1 upkeep
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_fallen_empire_spiritualist = yes }
		}
		text = job_fe_acolyte_farm_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = {
				is_fallen_empire_spiritualist = no
				is_gestalt = no
			}
		}
		text = job_farmer_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_agri_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_aqu_angler_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_aquaculture_drone_effect_desc
	}

	upgrades = {
		"building_aquaponics_farm_2"
	}

	prerequisites = {
		"tech_aquaponics"
	}
}
