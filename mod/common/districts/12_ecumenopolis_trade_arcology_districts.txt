district_arcology_commercial = {
	base_buildtime = 600
	is_capped_by_modifier = no
	show_on_uncolonized = {
		et_uses_district_set_city_world = yes
		exists = from
		from = {
			is_gestalt = no
			is_fallen_empire = no
		}
	}
	potential = {
		et_uses_district_set_city_world = yes
		exists = owner
		owner = {
			is_gestalt = no
			is_fallen_empire = no
		}
	}
	conversion_ratio = 1
	convert_to = {
		district_city
		district_nexus
		district_hive
		district_arcology_housing
	}
	resources = {
		category = planet_districts
		cost = {
			minerals = 1000
			rare_crystals = 50
		}
		upkeep = {
			energy = 5
			rare_crystals = 1
		}
	}
	planet_modifier = {
		planet_housing_add = 10
		job_clerk_add = 6
		job_merchant_add = 2
	}
	triggered_planet_modifier = {
		potential = { colony_type = col_ecu_trade }
		modifier = {
			job_clerk_add = -2
			job_merchant_add = 2
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				OR = {
					has_valid_civic = civic_merchant_guilds
					has_valid_civic = civic_corporate_dominion
				}
			}
		}
		modifier = {
			job_clerk_add = -2
			job_merchant_add = 2
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_active_tradition = tr_mercantile_trickle_up_economics }
		}
		modifier = {
			job_clerk_add = 3
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_active_tradition = tr_mercantile_commercial_enterprise }
		}
		modifier = {
			job_merchant_add = 2
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_megacorp = yes }
		}
		modifier = {
			# having 1 clerk => manager and 1 merchant => executive is more fair on a large district, where it's less of an overall power bump
			job_clerk_add = -1
			job_merchant_add = -1
			job_manager_add = 1
			job_executive_add = 1
		}
	}
	triggered_desc = {
		text = job_clerk_effect_desc
	}
	triggered_desc = {
		text = job_merchant_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_megacorp = yes }
		}
		text = job_manager_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_megacorp = yes }
		}
		text = job_executive_effect_desc
	}
}
