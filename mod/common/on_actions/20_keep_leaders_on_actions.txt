# fired when the game starts
on_game_start = {
	events = {
		keep_leaders.13 # flag mod as installed
	}
}

# fired when a country finishes being integrated
# THIS = overlord
# FROM = subject
on_subject_integrated = {
	events = {
		keep_leaders.1
	}
}

# fired when a country is destroyed
# THIS = destroyed country
# FROM = optional, destroyer (country)
on_country_destroyed = {
	events = {
		keep_leaders.2
	}
}

# fired when a leader has died
# THIS = country
# FROM = leader
on_leader_death = {
	events = {
		keep_leaders.100
		keep_leaders_bonus_traits.20
	}
}

# fired when a leader has been fired
# THIS = country
# FROM = leader
on_leader_fired = {
	events = {
		keep_leaders_bonus_traits.20
	}
}
