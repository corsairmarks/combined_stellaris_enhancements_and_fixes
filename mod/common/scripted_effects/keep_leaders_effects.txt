	if = {
		limit = {
			is_civilian_leader = yes

			is_eligible_for_policy_full_citizenship_species = yes
		}
		set_full_citizenship_species = yes
	}
	else_if = {
		limit = {
			is_military_leader = yes
			can_be_military_leaders_species = no
			is_eligible_for_policy_full_military_service_species = yes
		}
		set_full_military_service_species = yes
	}
	set_minimum_standard_of_living_species = yes
	unassign_leader = THIS
	set_owner = event_target:root_country
}

# THIS = something with species
# event_target:root_country = country to set citizenship
set_full_citizenship_species = {
	species= {
		if = {
			limit = {
				event_target:root_country = { is_machine_empire = yes }
				NOR = {
					has_citizenship_type = {
						country = event_target:root_country
						type = citizenship_full_machine
					}
					has_citizenship_type = {
						country = event_target:root_country
						type = citizenship_assimilation
					}
				}
			}
			set_citizenship_type = {
				country = event_target:root_country
				type = citizenship_full_machine
				cooldown = no
			}
		}
		else_if = {
			limit = {
				NOR = {
					has_citizenship_type = {
						country = event_target:root_country
						type = citizenship_full
					}
					has_citizenship_type = {
						country = event_target:root_country
						type = citizenship_assimilation
					}
				}
			}
			set_citizenship_type = {
				country = event_target:root_country
				type = citizenship_full
				cooldown = no
			}
			if = {
				limit = {
					event_target:root_country = { has_valid_civic = civic_citizen_service }
					NOT = {
						has_military_service_type = {
							country = event_target:root_country
							type = military_service_full
						}
					}
				}
				set_military_service_type = {
					country = event_target:root_country
					type = military_service_full
					cooldown = no
				}
			}
		}
	}
}

# THIS = something with species
# event_target:root_country = country to ensure minimum standard of living
set_minimum_standard_of_living_species = {
	species= {
		if = {
			limit = {
				event_target:root_country = {
					OR = {
						is_machine_empire = yes
						is_hive_empire = yes
					}
				}
				NOR = {
					has_living_standard = {
						country = event_target:root_country
						type = living_standard_hive_mind
					}
					has_living_standard = {
						country = event_target:root_country
						type = living_standard_ego_assimilation
					}
				}
			}
			set_living_standard = {
				country = event_target:root_country
				type = living_standard_hive_mind
			}
		}
		else_if = {
			limit = {
				event_target:root_country = { is_regular_empire = yes }
				NOR = {
					has_citizenship_type = {
						country = event_target:root_country
						type = citizenship_assimilation
					}
					AND = {
						has_citizenship_type = {
							country = event_target:root_country
							type = citizenship_purge
						}
						has_living_standard = {
							country = event_target:root_country
							type = living_standard_none
						}
						has_purge_type = {
							country = event_target:root_country
							type = purge_necrophage
						}
					}
					event_target:root_country = {
						is_robotic_species = yes
						has_full_ai_rights = no
					}
				}
			}
			if = {
				limit = { event_target:root_country = { has_valid_civic = civic_shared_burden } }
				set_living_standard = {
					country = event_target:root_country
					type = living_standard_shared_burden
					cooldown = no
				}
			}
			else_if = {
				limit = { event_target:root_country = { is_materialist = yes } }
				set_living_standard = {
					country = event_target:root_country
					type = living_standard_academic_privilege
					cooldown = no
				}
			}
			else_if = {
				limit = { event_target:root_country = { is_authoritarian = yes } }
				set_living_standard = {
					country = event_target:root_country
					type = living_standard_stratified
					cooldown = no
				}
			}
			else_if = {
				limit = {
					has_citizenship_type = {
						country = event_target:root_country
						type = citizenship_full
					} 
				}
				set_living_standard = {
					country = event_target:root_country
					type = living_standard_normal
					cooldown = no
				}
			}
			else = {
				set_living_standard = {
					country = event_target:root_country
					type = living_standard_subsistence
					cooldown = no
				}
			}
		}
	}
}

# THIS = leader
# event_target:prospective_owner = the country scope that will potentially become the owner of the leader
#    note - owner_species is used to compare for leader_not_same_species_accumulator
# event_target:variable_storage = the scope that has these 5 variables:
# * leader_organic_accumulator
# * leader_organic_noncyborg_accumulator
# * leader_not_same_species_accumulator
# * leader_necrophagable_accumulator
# * leader_accumulator
accumulate_leader = {
	if = {
		limit = { is_organic_species = yes }
		event_target:variable_storage = {
			change_variable = {
				which = leader_organic_accumulator
				value = 1
			}
		}
		if = {
			limit = { is_cyborg_leader = no }
			event_target:variable_storage = {
				change_variable = {
					which = leader_organic_noncyborg_accumulator
					value = 1
				}
			}
		}
	}
	if = {
		limit = { NOT = { is_same_species = event_target:prospective_owner.owner_species } }
		event_target:variable_storage = {
			change_variable = {
				which = leader_not_same_species_accumulator
				value = 1
			}
		}
	}
	if= {
		limit = {
			is_robotic_species = no
			species = {
				NOR = {
					is_sapient = no
					is_same_species = event_target:prospective_owner.owner_species
					has_slavery_type = {
						type = slavery_livestock
						country = event_target:prospective_owner
					}
				}
				species_has_happiness_with_owner = event_target:prospective_owner
			}
