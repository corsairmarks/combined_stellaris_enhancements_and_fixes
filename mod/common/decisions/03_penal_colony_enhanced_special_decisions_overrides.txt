decision_penal_colony = {
	owned_planets_only = yes
	prerequisites = {
		"tech_penal_colonies"
	}
	resources = {
		category = decisions
		cost = {
			unity = 1000
		}
	}
	potential = {
		is_capital = no
		NOR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
		}
		is_artificial = no
		owner = {
			is_regular_empire = yes
			is_fallen_empire = no
			NOT = { any_owned_planet = { has_modifier = penal_colony } }
			OR = {
				is_ai = no
				num_owned_planets > 5
			}
		}
	}
	allow = {
		planet_size >= 15
	}
	effect = {
		add_modifier = { modifier = penal_colony }
		hidden_effect = {
			set_colony_type = col_penal
			convert_capital_to_penal_capital = yes
			validate_planet_buildings_and_districts = yes
		}
	}
	ai_weight = {
		weight = 2
		modifier = {
			factor = 2
			exists = owner
			owner = { has_valid_civic = civic_police_state }
		}
		modifier = {
			factor = 2
			exists = owner
			owner = { is_authoritarian = yes }
		}
		modifier = {
			factor = 0.25
			exists = owner
			owner = { is_egalitarian = yes }
		}
		modifier = {
			factor = 2
			is_planet_class = pc_nuked
			exists = owner
			owner = {
				NOR = {
					ideal_planet_class = pc_nuked
					has_trait = trait_pc_nuked_preference
					has_trait = trait_survivor
					has_trait = trait_plantoid_radiotrophic
					has_trait = trait_harvested_radiotrophic
					has_trait = trait_lithoid_radiotrophic
					has_trait = trait_advanced_radiotrophic
					has_trait = trait_presapient_irradiated
				}
			}
		}
		modifier = {
			factor = 1.5
			has_modifier = bleak_planet
		}
		modifier = {
			factor = 0.1
			exists = owner
			habitability = {
				who = owner_main_species
				value >= 0.8
			}
		}
		modifier = {
			factor = 0.4
			exists = owner
			habitability = {
				who = owner_main_species
				value < 0.8
			}
			habitability = {
				who = owner_main_species
				value >= 0.6
			}
		}
		modifier = {
			factor = 0.7
			exists = owner
			habitability = {
				who = owner_main_species
				value < 0.6
			}
			habitability = {
				who = owner_main_species
				value >= 0.4
			}
		}
	}
}
