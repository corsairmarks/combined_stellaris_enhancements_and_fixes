# Modified:
# * homicidals are not blanket-denied from assimilating (potential)
# * non-gestalt empires with tech_sapient_ai are allowed to (de)assimilate machine units
# * cybernetic/synthetic assimilation can be restricted (with Civic: Organic Zealots)
citizenship_assimilation = {
	is_assimilation = yes
	potential = {
		OR = {
			has_synthethic_dawn = yes
			has_utopia = yes
		}
		exists = from
		from = {
			OR = {
				# driven assimilator cybernetic assimilation
				AND = {
					has_valid_civic = civic_machine_assimilator
					prev = {
						NOT = { has_trait = trait_cybernetic }
						is_robotic_species = no
					}
				}
				# machine synthetic assimilation (includes determined exterminators)
				AND = {
					is_machine_empire = yes
					prev = { is_robotic_species = yes }
				}
				AND = {
					is_hive_empire = yes
					NOT = { has_valid_civic = civic_hive_devouring_swarm }
					OR = {
						has_tradition = tr_genetics_creation # hive assimilation
						has_tradition = tr_cybernetics_transubstantiation_synthesis # cybernetic assimilation
					}
				}
				# non-gestalts
				AND = {
					is_gestalt = no
					NOT = { has_valid_civic = civic_fanatic_purifiers }
					OR = {
						# machine deassimilation
						AND = {
							has_technology = tech_sapient_ai
							prev = { has_trait = trait_machine_unit }
						}
						has_country_flag = synth_assimilation # synthetic assimilation
						has_active_tradition = tr_cybernetics_transubstantiation_synthesis # cybernetic assimilation
						has_active_tradition = tr_psionics_psionic_assimilation # psionic assimilation
						has_active_tradition = tr_genetics_creation # hive deassimilation
					}
				}
			}
		}
	}
	allow = {
		hidden_trigger = { exists = from }
		custom_tooltip = {
			fail_text = MUST_NOT_BE_DOMINANT_SPECIES
			NOT = { is_exact_same_species = from }
		}
		if = {
			limit = { is_lonely_hive_mind_pop = yes }
			custom_tooltip = {
				fail_text = lonely_hive_mind_pops_kill_themselves_citizenship_assimilation
				from = {
					OR = {
						has_tradition = tr_genetics_creation
						has_tradition = tr_cybernetics_transubstantiation_synthesis
						has_tradition = tr_cybernetics_assimilator_inevitable_assimilation
					}
				}
			}
		}
		if = {
			limit = { from = { is_gestalt = yes } }
			custom_tooltip = {
				fail_text = disconnected_machine_pops_collapse_citizenship
				is_disconnected_machine_pop = no
			}
		}
		else = {
			custom_tooltip = {
				fail_text = disconnected_machine_pops_collapse_without_ai_citizenship
				from = {
					OR = {
						prev = { is_disconnected_machine_pop = no }
						has_country_flag = synth_assimilation
						has_technology = tech_sapient_ai
					}
				}
			}
			if = {
				limit = {
					has_trait = trait_machine_unit
					from = { has_technology = tech_sapient_ai }
				}
				custom_tooltip = {
					fail_text = ROBOTS_ARE_OUTLAWED
					from = { has_policy_flag = robots_allowed }
				}
				custom_tooltip = {
					fail_text = AI_IS_OUTLAWED_POP
					from = { NOT = { has_policy_flag = ai_outlawed } }
				}
			}
		}
		if = {
			limit = {
				has_global_flag = civic_organic_zealots_installed
				from = {
					is_machine_empire = no
					OR = {
						# machine deassimilation
						AND = {
							is_gestalt = no
							has_technology = tech_sapient_ai
							prev = { has_trait = trait_machine_unit }
						}
						has_country_flag = synth_assimilation # synthetic assimilation (regular)
						# cybernetic assimilation (regular/hive)
						AND = {
							has_active_tradition = tr_cybernetics_transubstantiation_synthesis
							prev = {
								is_robotic_species = no
								NOT = { has_trait = trait_cybernetic }
							}
						}
					}
				}
			}
			# TODO: needs more refining if I want to enable multiple assimilation types!
			custom_tooltip = {
				fail_text = lost_robot_policy_war_fail
				from = { NOT = { has_country_flag = lost_robot_policy_war } }
			}
			custom_tooltip = {
				fail_text = is_organic_zealots_fail
				from = { NOT = { has_valid_civic = civic_organic_zealots } }
			}
			custom_tooltip = {
				fail_text = is_overlord_organic_zealots_fail
				from = {
					NAND = {
						exists = overlord
						overlord = { has_valid_civic = civic_organic_zealots }
					}
				}
			}
		}
		if = {
			limit = { from = { is_machine_empire = yes } }
			if = {
				limit = { from = { has_valid_civic = civic_machine_assimilator } }
				custom_tooltip = {
					fail_text = CYBORG_SPECIES_MUST_BE_ORGANIC
					NOT = { has_trait = trait_cybernetic }
				}
				OR = {
					NOT = { has_trait = trait_hive_mind }
					from = { has_active_tradition = tr_cybernetics_assimilator_inevitable_assimilation }
				}
			}
			if = {
				limit = { is_robotic_species = yes }
				custom_tooltip = {
					fail_text = requires_tradition_tr_synthetics_adopt
					from = { has_active_tradition = tr_synthetics_adopt_machine }
				}
			}
		}
		else_if = {
			limit = { from = { is_hive_empire = yes } }
			if = {
				limit = { from = { has_active_tradition = tr_genetics_creation } }
				custom_tooltip = {
					fail_text = HIVE_SPECIES_MUST_BE_NON_HIVE
					NOT = { has_trait = trait_hive_mind }
				}
			}
			if = {
				limit = { from = { has_active_tradition = tr_cybernetics_transubstantiation_synthesis } }
				custom_tooltip = {
					fail_text = HIVE_SPECIES_MUST_BE_NON_HIVE_CYBORG
					NAND = {
						has_trait = trait_hive_mind
						has_trait = trait_cybernetic
					}
				}
			}
			custom_tooltip = {
				fail_text = MACHINE_SPECIES_MACHINE
				is_robotic_species = no
			}
		}
		else_if = {
			limit = {
				is_disconnected_machine_pop = no # excluded so I don't have to write lots of exceptions
				from = { is_gestalt = no }
			}
			# cybernetic assimilation
			if = {
				limit = { from = { has_active_tradition = tr_cybernetics_transubstantiation_synthesis } }
				custom_tooltip = {
					fail_text = CYBORG_SPECIES_MUST_BE_ORGANIC
					OR = {
						has_trait = trait_hive_mind
						is_robotic_species = no
						NOT = { has_trait = trait_cybernetic }
					}
				}
			}
			# psionic assimilation
			else_if = {
				limit = { from = { has_active_tradition = tr_psionics_psionic_assimilation } }
				custom_tooltip = {
					fail_text = TRANSCENDING_SPECIES_MUST_BE_ORGANIC
					is_robotic_species = no
				}
				custom_tooltip = {
					fail_text = TRANSCENDING_SPECIES_IS_ALREADY_PSIONIC
					NOT = { has_trait = trait_psionic }
				}
				custom_tooltip = {
					fail_text = TRANSCENDING_SPECIES_CANT_BE_HIVE
					NOT = { has_trait = trait_hive_mind }
				}
			}
			# hive deassimilation
			else_if = {
				limit = { from = { has_active_tradition = tr_genetics_creation } }
				custom_tooltip = {
					fail_text = CAN_ONLY_ASSIMILATE_HIVES
					has_trait = trait_hive_mind
				}
			}
		}
	}
	ai_will_do = {
		factor = 100
		modifier = {
			factor = 0
			# modified so terminator AIs can assimilate other machines/robots
			from = {
				OR = {
					AND = {
						has_valid_civic = civic_machine_terminator
						NAND = {
							has_active_tradition = tr_synthetics_adopt_machine
							prev = { is_robotic_species = yes }
						}
					}
					has_valid_civic = civic_hive_devouring_swarm
					has_valid_civic = civic_fanatic_purifiers
				}
			}
		}
	}
}
