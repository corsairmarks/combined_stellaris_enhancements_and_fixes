# compared to a bureaucrat w/byzantine bureaucracy, produces +1 unity, +2 amenities, +1 stability, -10 crime
magistrate = {
	category = ruler
	condition_string = RULER_JOB_TRIGGER
	building_icon = building_bureaucratic_3
	clothes_texture_index = 3
	icon = bureaucrat

	possible_pre_triggers = {
		has_owner = yes
		is_enslaved = no
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	possible_precalc = can_fill_ruler_job

	resources = {
		category = planet_administrators
		produces = {
			unity = 6
		}
		produces = {
			trigger = {
				exists = owner
				owner = { has_valid_civic = civic_police_state }
			}
			unity = 1
		}
		produces = {
			trigger = {
				exists = owner
				owner = { has_active_tradition = tr_domination_judgment_corps }
			}
			unity = 1
		}
		upkeep = {
			trigger = {
				exists = planet
				planet = { has_modifier = anticrime_campaign }
			}
			energy = 2
		}
		upkeep = {
			consumer_goods = 2
		}
	}

	planet_modifier = {
		planet_stability_add = 2
		planet_crime_add = -10
	}
	triggered_planet_modifier = {
		potential = { always = yes }
		planet_amenities_add = 5
		mult = value:scripted_modifier_mult|MODIFIER|pop_job_amenities_mult|
	}
	triggered_country_modifier = {
		potential = {
			exists = owner
			owner = { has_valid_civic = civic_efficient_bureaucracy }
		}
		country_edict_fund_add = 2
		mult = value:job_total_output_modifier|JOB|magistrate|RESOURCE|unity|
	}
	triggered_planet_modifier = {
		potential = {
			exists = planet
			planet = { has_modifier = anticrime_campaign }
		}
		modifier = {
			planet_crime_add = -4 # ratio: 40% of base reduction
		}
	}

	weight = {
		weight = @ruler_job_weight
		mult = value:job_weights_modifier|JOB|magistrate|RESOURCE|unity|
		mult = value:scripted_modifier_job_weight_mult|MODIFIER|pop_job_amenities_mult|
		modifier = {
			factor = 0.2
			has_citizenship_rights = no
			is_robotic_species = no
		}
		modifier = {
			factor = 1.1
			is_pop_category = ruler
			is_unemployed = yes
		}
		modifier = {
			factor = 5
			has_job = magistrate
		}
		modifier = {
			factor = 0
			OR = {
				has_trait = trait_presapient_proles
				can_think = no
			}
		}
		# crisis purge
		modifier = {
			factor = 0.01
			exists = planet
			exists = planet.controller
			planet.controller = {
				OR = {
					is_country_type = swarm
					is_country_type = ai_empire
				}
			}
			OR = { # contingency won't purge itself
				NOT = { exists = event_target:custodian_bot }
				AND = {
					exists = event_target:custodian_bot
					NOT = { species = { is_same_value = event_target:custodian_bot } }
				}
			}
		}
	}
}
