namespace = terraforming_stations_aesthetic

# flag mod as active
# triggered by on_game_start and on_single_player_save_game_load
event = {
	id = terraforming_stations_aesthetic.13
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		NOT = { has_global_flag = terraforming_stations_aesthetic_installed }
	}
	immediate = {
		set_global_flag = terraforming_stations_aesthetic_installed
	}
}

# set up terraform station and graphics
# triggered by on_terraforming_begun
# THIS = planet
# FROM = country, terraformer
planet_event = {
	id = terraforming_stations_aesthetic.1
	is_triggered_only = yes
	hide_window = yes
	trigger = {
		NOT = { solar_system = { any_orbital_station = { has_fleet_flag = terraform_station@PREVPREV } } }
	}
	immediate = {
		save_event_target_as = terraform_planet
		create_fleet = {
			set_take_point = no
			settings = {
				can_change_composition = no
				uses_naval_capacity = no
				ai_ignore_strength = yes
				can_change_leader = no
				garrison = yes
				spawn_debris = no
			}
			effect = {
				set_name = NAME_fleet_terraform_station
				set_owner = FROM
				set_fleet_flag = terraform_station@event_target:terraform_planet
				set_location = {
					target = event_target:terraform_planet
					distance = 0
					angle = random
				}
				create_ship = {
					graphical_culture = FROM
					prefix = no
					random_existing_design = terraform_station
					effect = {
						set_name = NAME_ship_terraform_station
					}
				}
			}
		}
		planet_event = {
			id = terraforming_stations_aesthetic.3
			days = 30
		}
	}
}

# clean up terraform station and graphics
# triggered by on_terraforming_complete and terraforming_stations_aesthetic.3
# THIS = planet
# FROM = country, terraformer
planet_event = {
	id = terraforming_stations_aesthetic.2
	is_triggered_only = yes
	hide_window = yes
	trigger = {
		solar_system = { any_orbital_station = { has_fleet_flag = terraform_station@PREVPREV } }
	}
	immediate = {
		solar_system = {
			every_orbital_station = {
				limit = { has_fleet_flag = terraform_station@PREVPREV }
				delete_fleet = THIS
			}
			every_ambient_object = {
				limit = { has_ambient_object_flag = terraform_station@PREVPREV }
				destroy_ambient_object = THIS
			}
			every_megastructure = {
				limit = { has_megastructure_flag = terraform_station@PREVPREV }
				delete_megastructure = THIS
			}
		}
	}
}

# check to see if terraforming has stopped - if so, fire the cleanup event, otherwise check again in a month
# triggered by terraforming_stations_aesthetic.1 and terraforming_stations_aesthetic.3
# THIS = planet
# FROM = country, terraformer
planet_event = {
	id = terraforming_stations_aesthetic.3
	is_triggered_only = yes
	hide_window = yes
	trigger = {
		solar_system = { any_orbital_station = { has_fleet_flag = terraform_station@PREVPREV } }
	}
	immediate = {
		if = {
			limit = { is_terraforming = yes }
			planet_event = {
				id = terraforming_stations_aesthetic.3
				days = 30
				scopes = {
					from = FROM
				}
			}
		}
		else = {
			planet_event = {
				id = terraforming_stations_aesthetic.2
				scopes = {
					from = FROM
				}
			}
		}
	}
}
