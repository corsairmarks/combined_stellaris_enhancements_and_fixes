# a copy of the default Full Military Service rights, with extra logic added so Battle Thralls are eligible
military_service_full = {
	token = military_service_full
	potential = {
		if = {
			limit = { FROM = { has_authority = auth_machine_intelligence } }
			OR = {
				has_trait = trait_machine_unit
				has_trait = trait_mechanical
				has_trait = trait_cybernetic
			}
		}
	}
	allow = {
		hidden_trigger = { exists = FROM }
		custom_tooltip = {
			fail_text = MACHINE_NO_PURE_ORGANIC_SOLDIERS
			if = {
				limit = { FROM = { has_authority = auth_machine_intelligence } }
				OR = {
					has_trait = trait_mechanical
					has_trait = trait_machine_unit
					AND = {
						has_trait = trait_cybernetic
						FROM = { has_civic = civic_machine_assimilator }
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = is_undesirables_tooltip
			NOR = {
				has_citizenship_type = {
					country = FROM
					type = citizenship_purge_machine
				}
				has_citizenship_type = {
					country = FROM
					type = citizenship_purge
				}
			}
		}
		custom_tooltip = {
			fail_text = is_slaves_tooltip
			OR = {
				NOT = {
					has_citizenship_type = {
						country = FROM
						type = citizenship_slavery
					}
				}
				has_slavery_type = {
					country = FROM
					type = slavery_military
				}
			}
		}
		custom_tooltip = {
			fail_text = is_servitude_tooltip
			NOT = {
				has_citizenship_type = {
					country = FROM
					type = citizenship_robot_servitude
				}
			}
		}
		custom_tooltip = {
			fail_text = too_alien
			OR = {
				is_same_species = FROM
				has_species_flag = allow_military_leaders_for_xenophobe # surprise flag that will help us later
				FROM = { is_xenophobe = no }
			}
		}
		custom_tooltip = {
			fail_text = necrophage_only_allows_necrophage_leaders
			OR = {
				is_same_species = FROM
				has_species_flag = allow_military_leaders_for_necrophage # surprise flag that will help us later
				FROM = {
					NAND = {
						has_origin = origin_necrophage
						has_trait = trait_necrophage
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = HIVE_MIND_SPECIES_CITIZENS_ONLY
			OR = {
				NOT = { FROM = { has_authority = auth_hive_mind } }
				has_trait = trait_hive_mind
			}
		}
		custom_tooltip = {
			fail_text = NON_MACHINE_SPECIES_NEED_PREREQ_TECH
			if = {
				limit = {
					OR = {
						has_trait = trait_mechanical
						has_trait = trait_machine_unit
					}
				}
				FROM = {
					OR = {
						has_authority = auth_machine_intelligence
						has_technology = tech_synthetic_leaders
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = nerve_stapled_cannot_lead_armies
			NOR = {
				is_sapient = no
				has_trait = trait_syncretic_proles
				has_trait = trait_nerve_stapled
			}
		}
	}
	triggered_pop_modifier = {
		potential = {
			has_slavery_type = { type = slavery_military }
		}
		modifier = {
			pop_cat_slave_happiness = 0.1
		}
	}
	ai_will_do = {
		factor = 5
		modifier = {
			factor = 0
			FROM = { has_valid_ai_personality = no }
		}
		modifier = {
			factor = 0
			FROM = {
				has_valid_ai_personality = yes
				NOT = { has_ai_personality_behaviour = multispecies }
			}
		}
	}
}
